### AND OR y XOR (emparejamiento) 2

x <- c(FALSE, FALSE, TRUE, TRUE,  TRUE)
y <- c(FALSE, TRUE,  TRUE, FALSE, TRUE)

y | (x & y)   ---> FALSE TRUE  TRUE  FALSE TRUE
x & xor(x, y) ---> FALSE FALSE FALSE TRUE  FALSE
y & (x | y)   ---> FALSE TRUE  TRUE  FALSE TRUE
!y | xor(x, y)---> TRUE  TRUE  FALSE TRUE  FALSE
